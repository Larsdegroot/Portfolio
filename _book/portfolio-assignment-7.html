5.4 4. Store the three tables as separate (so six in total) .csv and .rds files.

for (x in c("flu", "dengue", "gapminder")) {
  filecsv <- paste0(paste0(here("7_data", x)), "_tidy.csv") 
  filerds <- paste0(paste0(here("7_data", x)), "_tidy.rds")
  write.csv(get(x), file = filecsv)
  saveRDS(get(x), file = filerds)
}
